void showAll(queue <int> q)

    while (!q.empty())

    }

{
    queue <int> q;
    q.push(10);
    q.push(20);
    q.push(30);

    cout << "Show all queue elements : ";
    showAll(q);

    cout << "\nq.back() : " << q.back();

    q.pop();
    cout << "Show all queue elements : ";
    showAll(q);

    cout << "\nq.push()\n";
    q.push(50);
    cout << "Show all queue elements : ";
    showAll(q);

    return 0;
}
